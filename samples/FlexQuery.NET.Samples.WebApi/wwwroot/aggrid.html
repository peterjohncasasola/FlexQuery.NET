// 1. Parse AG Grid SSRM request to QueryOptions
var options = request.ToQueryOptions();
options.IncludeCount = true;

// 2. Execute against EF Core
var result = await _db.Customers
    .AsNoTracking()
    .FlexQueryAsync<Customer>(options);

// 3. Return AG Grid-compatible response
return Ok(result.ToAgGridServerSideResponse(request));
