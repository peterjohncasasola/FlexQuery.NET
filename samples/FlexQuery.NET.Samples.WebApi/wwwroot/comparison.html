[HttpGet]
public async Task<IActionResult> GetCustomers(
    [FromQuery] FlexQueryParameters p)
{
    var result = await connection
        .FlexQueryAsync<Customer>(p, opt =>
    {
        opt.Dialect = new SqliteDialect();
        opt.MappingRegistry = _registry;
    });

    return Ok(result);
}
