// 1. Parse Kendo DataSource request to QueryOptions
var options = request.ToQueryOptions();
options.IncludeCount = true;

// 2. Execute against EF Core
var result = await _db.Customers
    .AsNoTracking()
    .FlexQueryAsync<Customer>(options);

// 3. Return Kendo-compatible response
return Ok(new { data = result.Data, total = result.TotalCount });
